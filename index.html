Ejemplos de JavaScript

Control de flujo y manejo de errores

if...else

Ejemplo:


let edad = 18;
if (edad >= 18) {
    console.log("Eres mayor de edad");
} else {
    console.log("Eres menor de edad");
}

switch

Ejemplo:


let diaDeLaSemana = 3;
switch (diaDeLaSemana) {
    case 1:
        console.log("Lunes");
        break;
    case 2:
        console.log("Martes");
        break;
    case 3:
        console.log("Miércoles");
        break;
    default:
        console.log("Día no válido");
}

try/catch/throw

Ejemplo:


function dividir(a, b) {
    try {
        if (b === 0) {
            throw new Error("No se puede dividir por cero");
        }
        return a / b;
    } catch (error) {
        console.error(error.message);
    }
}

console.log(dividir(10, 2)); // Resultado: 5
console.log(dividir(10, 0)); // Mostrará el mensaje de error

Objetos Error

Ejemplo:


try {
    throw new Error("Este es un error personalizado");
} catch (error) {
    console.error("Ocurrió un error:", error.message);
}

Bucles e iteración

for

Ejemplo:


for (let i = 0; i < 5; i++) {
    console.log(i);
}

while

Ejemplo:


let contador = 0;
while (contador < 5) {
    console.log(contador);
    contador++;
}

do...while

Ejemplo:


let contador = 0;
do {
    console.log(contador);
    contador++;
} while (contador < 5);

Funciones

Definir funciones

Ejemplo:


// Definir una función para sumar dos números
function sumar(a, b) {
    return a + b;
}

// Definir una función para calcular el área de un rectángulo
function calcularArea(base, altura) {
    return base * altura;
}

Llamar funciones

Ejemplo:


// Llamar la función sumar y mostrar el resultado
let resultadoSuma = sumar(5, 3);
console.log("Resultado de la suma:", resultadoSuma);

// Llamar la función calcularArea y mostrar el resultado
let areaRectangulo = calcularArea(10, 5);
console.log("Área del rectángulo:", areaRectangulo);

Ejemplos de Operadores en JavaScript

Asignación

El operador de asignación = se usa para asignar un valor a una variable.

let x = 10;

Operadores de asignación compuestos:


let y = 5;
y += 3; // Equivalente a y = y + 3
y -= 2; // Equivalente a y = y - 2

Comparación

Operadores de comparación se usan para comparar dos valores.


let a = 5;
let b = 10;
let c = 5;
let result1 = (a == c); // true
let result2 = (a == b); // false
let result3 = (a != b); // true

Operadores Aritméticos

Operadores aritméticos realizan operaciones matemáticas sobre números.


let num1 = 10;
let num2 = 3;
let sum = num1 + num2;        // 13
let difference = num1 - num2; // 7
let product = num1 * num2;    // 30
let quotient = num1 / num2;   // 3.33
let remainder = num1 % num2;  // 1

Números y fechas

Objeto Number

Ejemplo:


let num = 10;
console.log(Number.isInteger(num)); // true
console.log(num.toFixed(2)); // "10.00"

Objeto Math

Ejemplo:


console.log(Math.sqrt(16)); // 4
console.log(Math.floor(4.6)); // 4
console.log(Math.random()); // número aleatorio entre 0 y 1

Objeto Date

Ejemplo:


let fechaActual = new Date();
console.log(fechaActual.getFullYear()); // año actual
console.log(fechaActual.getMonth()); // mes actual (0-11)
console.log(fechaActual.getDate()); // día del mes

Formateo de texto

Cadenas literales

Ejemplo:


let nombre = "Juan";
let mensaje = `Hola, ${nombre}!`;
console.log(mensaje); // "Hola, Juan!"

Objeto String

Ejemplo:


let texto = "JavaScript";
console.log(texto.length); // longitud del texto
console.log(texto.toUpperCase()); // "JAVASCRIPT"
console.log(texto.substring(0, 4)); // "Java"

let numeros = new Int32Array([1, 2, 3, 4, 5]);

DOM (Document Object Model)

Encontrar elementos HTML por ID

Ejemplo:


let elementoPorId = document.getElementById('miElemento');
console.log(elementoPorId);

Encontrar elementos HTML por nombre de etiqueta

Ejemplo:


let elementosPorEtiqueta = document.getElementsByTagName('div');
console.log(elementosPorEtiqueta);

Encontrar elementos HTML por nombre de clase

Ejemplo:


let elementosPorClase = document.getElementsByClassName('ejemplo');
console.log(elementosPorClase);

Encontrar elementos HTML mediante selectores CSS

Ejemplo:


let elementoPorSelector = document.querySelector('#miElemento');
console.log(elementoPorSelector);

Encontrar elementos HTML por colecciones de objetos HTML

Ejemplo:


let todosLosDivs = document.querySelectorAll('div');
console.log(todosLosDivs);
